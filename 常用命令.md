### dll插件快速部署
```powershell
Copy-Item -Path # 定义路径
$buildDir = "e:\Dev\mini-program\Browser-Storage-Backup\game_mod_develop\lua\VaultSave_Phantom_Portal\Plugin-making\build\Release"
$gameDir = "E:\SteamLibrary\steamapps\common\RESIDENT EVIL requiem BIOHAZARD requiem"
$pluginDir = "$gameDir\reframework\plugins"
$dllName = "VaultSave_RE9_PhantomPortal.dll" # 新名称

# 清理旧插件（可选）
if (Test-Path "$pluginDir\VaultCore_RE9_PhantomPortal.dll") {
    Remove-Item "$pluginDir\VaultCore_RE9_PhantomPortal.dll" -Force
    Write-Host "已清理旧版插件"
}

# 部署新插件
Copy-Item -Path "$buildDir\$dllName" -Destination "$pluginDir" -Force
Write-Host "已部署 $dllName"
```
